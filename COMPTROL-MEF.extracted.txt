-- Extensiones recomendadas
CREATE EXTENSION IF NOT EXISTS "pgcrypto"; -- gen_random_uuid()
-- Tenants (SaaS B2B)
CREATE TABLE tenants (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
name VARCHAR(160) NOT NULL,
slug VARCHAR(80) UNIQUE NOT NULL,
created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
-- Locations
CREATE TABLE locations (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
tenant_id UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
name VARCHAR(120) NOT NULL,
code VARCHAR(40),
country VARCHAR(80),
city VARCHAR(80),
address_line1 VARCHAR(160),
address_line2 VARCHAR(160),
is_active BOOLEAN NOT NULL DEFAULT TRUE,
created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
UNIQUE (tenant_id, name),
UNIQUE (tenant_id, code)
);
-- Users
CREATE TABLE users (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
tenant_id UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
email VARCHAR(190) NOT NULL,
full_name VARCHAR(140) NOT NULL,
role VARCHAR(50) NOT NULL CHECK (role IN ('super_admin','it_admin','asset_manager','security_analyst','auditor','employee')),
mfa_enabled BOOLEAN NOT NULL DEFAULT FALSE,
sso_provider VARCHAR(40), -- azure_ad, google, local
status VARCHAR(30) NOT NULL DEFAULT 'active' CHECK (status IN ('active','inactive','suspended')),
last_login_at TIMESTAMPTZ,
created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
UNIQUE (tenant_id, email)
);
-- Assets (CMDB source of truth)
CREATE TABLE assets (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
tenant_id UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
asset_tag VARCHAR(80) NOT NULL,
serial_number VARCHAR(120),
hostname VARCHAR(120),
ip_address INET,
mac_address MACADDR,
asset_type VARCHAR(40) NOT NULL CHECK (asset_type IN ('laptop','desktop','server','network','iot','ot','mobile','virtual_machine','cloud_resource','other')),
vendor VARCHAR(80),
model VARCHAR(120),
operating_system VARCHAR(120),
status VARCHAR(30) NOT NULL DEFAULT 'in_use' CHECK (status IN ('in_use','in_stock','repair','retired','disposed','lost')),
criticality VARCHAR(20) NOT NULL DEFAULT 'medium' CHECK (criticality IN ('low','medium','high','critical')),
owner_user_id UUID REFERENCES users(id) ON DELETE SET NULL,
location_id UUID REFERENCES locations(id) ON DELETE SET NULL,
purchase_date DATE,
warranty_end_date DATE,
depreciation_end_date DATE,
purchase_cost NUMERIC(14,2) DEFAULT 0,
current_book_value NUMERIC(14,2) DEFAULT 0,
last_seen_at TIMESTAMPTZ, -- discovery heartbeat
source VARCHAR(30) NOT NULL DEFAULT 'manual' CHECK (source IN ('manual','discovery_active','discovery_passive','api_import')),
created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
UNIQUE (tenant_id, asset_tag),
UNIQUE (tenant_id, serial_number)
);
-- Software Licenses (SAM)
CREATE TABLE software_licenses (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
tenant_id UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
software_name VARCHAR(140) NOT NULL,
vendor VARCHAR(120),
license_type VARCHAR(40) NOT NULL CHECK (license_type IN ('per_user','per_device','concurrent','subscription','enterprise')),
total_seats INTEGER NOT NULL CHECK (total_seats >= 0),
assigned_seats INTEGER NOT NULL DEFAULT 0 CHECK (assigned_seats >= 0),
active_agents INTEGER NOT NULL DEFAULT 0 CHECK (active_agents >= 0),
inactive_agents INTEGER NOT NULL DEFAULT 0 CHECK (inactive_agents >= 0),
unit_cost NUMERIC(14,2) NOT NULL DEFAULT 0,
renewal_date DATE,
auto_renew BOOLEAN NOT NULL DEFAULT FALSE,
status VARCHAR(30) NOT NULL DEFAULT 'active' CHECK (status IN ('active','expiring','expired','suspended')),
created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
-- Audit Logs (inmutable / append-only)
CREATE TABLE audit_logs (
id BIGSERIAL PRIMARY KEY,
tenant_id UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
event_id UUID NOT NULL DEFAULT gen_random_uuid(),
actor_user_id UUID REFERENCES users(id) ON DELETE SET NULL,
entity_type VARCHAR(60) NOT NULL, -- assets, software_licenses, users...
entity_id UUID NOT NULL,
action VARCHAR(40) NOT NULL, -- create, update, delete, assign, discover...
before_data JSONB,
after_data JSONB,
metadata JSONB,
ip_address INET,
user_agent TEXT,
created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
UNIQUE(event_id)
);
-- Índices recomendados
CREATE INDEX idx_assets_tenant_status ON assets(tenant_id, status);
CREATE INDEX idx_assets_tenant_last_seen ON assets(tenant_id, last_seen_at DESC);
CREATE INDEX idx_assets_tenant_owner ON assets(tenant_id, owner_user_id);
CREATE INDEX idx_licenses_tenant_status ON software_licenses(tenant_id, status);
CREATE INDEX idx_audit_tenant_entity_created ON audit_logs(tenant_id, entity_type, entity_id, created_at DESC);
-- Trigger updated_at
CREATE OR REPLACE FUNCTION set_updated_at()
RETURNS TRIGGER AS $$
BEGIN
NEW.updated_at = NOW();
RETURN NEW;
END;
$$ LANGUAGE plpgsql;
CREATE TRIGGER trg_users_updated BEFORE UPDATE ON users
FOR EACH ROW EXECUTE FUNCTION set_updated_at();
CREATE TRIGGER trg_locations_updated BEFORE UPDATE ON locations
FOR EACH ROW EXECUTE FUNCTION set_updated_at();
CREATE TRIGGER trg_assets_updated BEFORE UPDATE ON assets
FOR EACH ROW EXECUTE FUNCTION set_updated_at();
CREATE TRIGGER trg_licenses_updated BEFORE UPDATE ON software_licenses
FOR EACH ROW EXECUTE FUNCTION set_updated_at();
-- Protección de inmutabilidad en audit_logs (no UPDATE/DELETE)
CREATE OR REPLACE FUNCTION prevent_audit_log_mutation()
RETURNS TRIGGER AS $$
BEGIN
RAISE EXCEPTION 'audit_logs is append-only';
END;
$$ LANGUAGE plpgsql;
CREATE TRIGGER trg_audit_no_update
BEFORE UPDATE ON audit_logs
FOR EACH ROW EXECUTE FUNCTION prevent_audit_log_mutation();
CREATE TRIGGER trg_audit_no_delete
BEFORE DELETE ON audit_logs
FOR EACH ROW EXECUTE FUNCTION prevent_audit_log_mutation();

import { IsEnum, IsIP, IsMACAddress, IsOptional, IsString, IsUUID, MaxLength } from 'class-validator';
export enum DiscoverySource {
ACTIVE = 'discovery_active',
PASSIVE = 'discovery_passive',
}
export class DiscoveryIngestDto {
@IsUUID()
tenantId!: string;
@IsOptional()
@IsString()
@MaxLength(120)
hostname?: string;
@IsOptional()
@IsIP()
ipAddress?: string;
@IsOptional()
@IsMACAddress()
macAddress?: string;
@IsOptional()
@IsString()
@MaxLength(120)
serialNumber?: string;
@IsString()
@MaxLength(40)
assetType!: string; // idealmente enum interno
@IsOptional()
@IsString()
@MaxLength(80)
vendor?: string;
@IsOptional()
@IsString()
@MaxLength(120)
model?: string;
@IsEnum(DiscoverySource)

}

// apps/api/src/modules/discovery/discovery.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { DiscoveryIngestDto } from './dto/discovery-ingest.dto';
/**
* Simulación de persistencia.
* En producción: repositorio (Prisma/TypeORM) + transacción + outbox event.
*/
@Injectable()
export class DiscoveryService {
private readonly logger = new Logger(DiscoveryService.name);
async ingestAsset(dto: DiscoveryIngestDto) {
// 1) Fingerprint para deduplicación (prioridad: serial > mac > hostname+ip)
const fingerprint =
dto.serialNumber?.trim() ||
dto.macAddress?.toLowerCase() ||
`${dto.hostname ?? 'unknown'}|${dto.ipAddress ?? 'unknown'}`;
// 2) Simular "upsert"
const simulatedAsset = {
id: crypto.randomUUID(),
tenantId: dto.tenantId,
assetTag: `AUTO-${Date.now()}`,
hostname: dto.hostname ?? null,
ipAddress: dto.ipAddress ?? null,
macAddress: dto.macAddress ?? null,
serialNumber: dto.serialNumber ?? null,
assetType: dto.assetType,
vendor: dto.vendor ?? null,
model: dto.model ?? null,
source: dto.source,
status: 'in_use',
lastSeenAt: new Date().toISOString(),
createdAt: new Date().toISOString(),
updatedAt: new Date().toISOString(),
fingerprint,
};
this.logger.log(`Asset discovered tenant=${dto.tenantId} fp=${fingerprint}`);
// 3) Simular evento de auditoría
const auditEvent = {
action: 'discover',
entityType: 'assets',
entityId: simulatedAsset.id,
afterData: simulatedAsset,
createdAt: new Date().toISOString(),
};
return { asset: simulatedAsset, auditEvent };
}
}

// apps/api/src/modules/discovery/discovery.controller.ts
import { Body, Controller, HttpCode, HttpStatus, Post } from '@nestjs/common';
import { DiscoveryIngestDto } from './dto/discovery-ingest.dto';
import { DiscoveryService } from './discovery.service';
@Controller('api/v1/assets')
export class DiscoveryController {
constructor(private readonly discoveryService: DiscoveryService) {}
/**
* Endpoint de ingesta de descubrimiento de activos (activo/pasivo).
* Recomendado proteger con JWT + mTLS/API key entre scanner -> API.
*/
@Post('discovery')
@HttpCode(HttpStatus.CREATED)
async ingestDiscovery(@Body() dto: DiscoveryIngestDto) {
const result = await this.discoveryService.ingestAsset(dto);
return {
success: true,
message: 'Asset discovery ingested successfully',
data: result.asset,
audit: result.auditEvent,
};
}
}

### 1.1 Tablas SQL (roles, permisos y alcance por recurso)
sql
-- Catálogo de permisos (acción + recurso)
CREATE TABLE permissions (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
resource VARCHAR(80) NOT NULL, -- assets, licenses, users, audit_logs...
action VARCHAR(40) NOT NULL, -- create, read, update, delete, approve, export
description TEXT,
created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
UNIQUE(resource, action)
);
-- Roles por tenant (evita rol global rígido)
CREATE TABLE roles (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
tenant_id UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
name VARCHAR(80) NOT NULL, -- IT_ADMIN, ASSET_MANAGER...
description TEXT,
is_system BOOLEAN NOT NULL DEFAULT FALSE,
created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
UNIQUE(tenant_id, name)
);
-- N:N roles-permissions
CREATE TABLE role_permissions (
role_id UUID NOT NULL REFERENCES roles(id) ON DELETE CASCADE,
permission_id UUID NOT NULL REFERENCES permissions(id) ON DELETE CASCADE,
effect VARCHAR(10) NOT NULL DEFAULT 'allow' CHECK (effect IN ('allow','deny')),
PRIMARY KEY(role_id, permission_id)
);
-- N:N users-roles
CREATE TABLE user_roles (
user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
role_id UUID NOT NULL REFERENCES roles(id) ON DELETE CASCADE,
assigned_by UUID REFERENCES users(id) ON DELETE SET NULL,
assigned_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
PRIMARY KEY(user_id, role_id)
);
-- Restricción de alcance (scope) opcional por ubicación/cost center
CREATE TABLE policy_scopes (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

### 3.1 Política recomendada
- Retry: 5 intentos
- Backoff exponencial + jitter: 5s, 15s, 45s, 120s, 300s
- Timeout por job scanner: 30s-60s
- Circuit breaker por segmento de red
- DLQ después de agotar reintentos
- Idempotency key: hash (tenant + serial/mac/ip+hostname)
### 3.2 Flujo
1) API recibe discovery -> guarda asset/upsert + outbox en misma tx
2) Publisher empuja a bus
3) Processor intenta enriquecer/normalizar
4) Si falla transitorio -> requeue con backoff
5) Si falla permanente -> DLQ + alerta SOC/IT Ops

- [ ] 4) CÓDIGO BASE: OUTBOX PUBLISHER (NestJS)
ts
// apps/api/src/modules/outbox/outbox.publisher.ts
import { Injectable, Logger } from '@nestjs/common';
import { DataSource } from 'typeorm';
@Injectable()
export class OutboxPublisher {
private readonly logger = new Logger(OutboxPublisher.name);
constructor(private readonly ds: DataSource) {}
// Ejecutar cada X segundos via cron interno o worker separado
async publishBatch(limit = 100): Promise<number> {
return this.ds.transaction(async (manager) => {
const rows = await manager.query(
`
SELECT *
FROM outbox_events
WHERE status = 'pending' AND next_attempt_at <= NOW()
ORDER BY created_at ASC
LIMIT $1
FOR UPDATE SKIP LOCKED
`,
[limit],
);
let published = 0;
for (const evt of rows) {
try {
// TODO: enviar a Kafka/Rabbit/SQS
// await this.bus.publish(evt.event_type, { id: evt.id, ...evt.payload });
await manager.query(
`UPDATE outbox_events
SET status='published', published_at=NOW(), attempts=attempts+1
WHERE id=$1`,
[evt.id],
);
published++;
} catch (err: any) {
const attempts = Number(evt.attempts) + 1;
const delaySec = Math.min(300, Math.pow(3, attempts)); // backoff simple
await manager.query(
`UPDATE outbox_events
SET attempts=$2,
status=CASE WHEN $2 >= 5 THEN 'failed' ELSE 'pending' END,
next_attempt_at=NOW() + ($3 || ' seconds')::interval,
last_error=$4
WHERE id=$1`,
[evt.id, attempts, delaySec, String(err?.message ?? err)],
);
if (attempts >= 5) {
await manager.query(
`INSERT INTO dead_letter_events
(original_message_id, tenant_id, consumer_name, event_type, payload, error, attempts)
VALUES ($1, $2, 'outbox_publisher', $3, $4::jsonb, $5, $6)`,
[
evt.id,
evt.tenant_id,
evt.event_type,
JSON.stringify(evt.payload),
String(err?.message ?? err),
attempts,
],
);
}
}
}
this.logger.log(`Outbox published=${published}/${rows.length}`);
return published;
});
}
}

-- Snapshot diario de KPIs (para dashboard rápido)
CREATE TABLE kpi_daily_snapshots (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

total_assets INTEGER NOT NULL DEFAULT 0,
inactive_licenses INTEGER NOT NULL DEFAULT 0,

UNIQUE (tenant_id, snapshot_date)

-- Alertas inteligentes (rule-based + score)
CREATE TABLE smart_alerts (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
tenant_id UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
alert_type VARCHAR(60) NOT NULL, -- license_idle, rogue_asset, no_agent, ewaste_stalled
severity VARCHAR(20) NOT NULL CHECK (severity IN ('low','medium','high','critical')),
title VARCHAR(180) NOT NULL,
description TEXT NOT NULL,
entity_type VARCHAR(40), -- asset, license, disposal_case
entity_id UUID,
risk_score NUMERIC(5,2), -- 0..1 o 0..100 (definir estándar)
status VARCHAR(20) NOT NULL DEFAULT 'open' CHECK (status IN ('open','acknowledged','resolved','dismissed')),
detected_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
due_at TIMESTAMPTZ,
resolved_at TIMESTAMPTZ,
metadata JSONB,
created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
CREATE INDEX idx_alerts_tenant_status_sev ON smart_alerts(tenant_id, status, severity, detected_at DESC);
